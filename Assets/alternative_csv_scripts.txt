using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using System;

public class CSVReader : MonoBehaviour
{
    public TextAsset textAssetData;
    [System.Serializable]
    public class QuestionCard{
        public string id;
        public string question;
        public string wrongOption1;
        public string wrongOption2;
        public string wrongOption3;
        public string wrongOption4;
        public string rightOption;
    }
    public class QuestionCardList{
        public QuestionCard[] questionCard;
    }

    public QuestionCardList myQuestionList = new QuestionCardList();
    void Start()
    {
        ReadCSV();
    }
    void ReadCSV(){
        string[] data = textAssetData.text.Split(new string[] {";", "\n"}, StringSplitOptions.None);
        int tableSize = data.Length / 6 - 1;
        myQuestionList.questionCard = new QuestionCard[tableSize];
        for(int i = 0; i< tableSize; i++){
            myQuestionList.questionCard[i] = new QuestionCard();
            myQuestionList.questionCard[i].id = data[6* (i+1)];
            myQuestionList.questionCard[i].wrongOption1 = data[6* (i+1)+1];
            myQuestionList.questionCard[i].wrongOption2 = data[6* (i+1)+2];
            myQuestionList.questionCard[i].wrongOption3 = data[6* (i+1)+3];
            myQuestionList.questionCard[i].wrongOption4 = data[6* (i+1)+4];
            myQuestionList.questionCard[i].rightOption = data[6* (i+1)+5];

        }
    }
}
